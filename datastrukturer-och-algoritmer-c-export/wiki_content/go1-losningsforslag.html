Algorithm max(list)
    pos <- first (list)
    max <- inspect(pos,list)
    while(pos≠end(list)) do
        if inspect(pos,list)> max then
            max <- inspect(pos,list)
        pos <- next(pos,list)
    return max

empty  konstrueras som list-empty
isempty  konstrueras som list-isempty
push(v, s) konstrueras som list-insert(v, list-end(s), s)
pop(s) konstrueras som list-remove(list-previous(list-end(s)), s)
top(s) konstrueras som list-inspect(list-previous(list-end(s)), s)

Top(Pop(Push(3, Push(6, Pop(Push(7, Push(4, Empty)))))))

Det finns 5 axiom i boken och det gäller att finna vilket axiom som man kan matcha formeln med.

  1. Axiom 3: Pop(Push(v, s)) = s. Låt v = 3 och s = Push(6, Pop(Push(7, Push(4, Empty)))) då får vi:
    Top(Push(6, Pop(Push(7, Push(4, Empty)))))
  2. Axiom 4: Top(Push(v, s)) = v. Låt v = 6 och s = Pop(Push(7, Push(4, Empty)))
    6

Dvs resultatet är värdet 6
